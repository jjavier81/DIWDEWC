1. Crea una web que contenga elementos p, div, h1, h2 y h3 con texto que cambiarán su contenido cuando se produzca un evento de la siguiente manera:

  1. Incluye una manera distinta de asociar la función de manejo de eventos para cada elemento.
  2. Además, escoge un evento de ratón distinto para cada elemento salvo para el elemento div que debe tener asociado el evento del menú secundario.
  3. Impide que el menú secundario haga la acción que tiene asociada por defecto.

Elemento h1

/*Evento: UN CLIC*/
document.getElementsByTagName('h1')[0].onclick = function() {
    document.getElementsByTagName('h1')[0].innerHTML = "Evento 'onclick' sobre elemento h1";
}

Elemento p

/*Evento: DOBLE CLIC*/
document.getElementsByTagName("p")[2].ondblclick = function(){
    document.getElementsByTagName("p")[2].innerHTML = "Evento 'ondblclick' sobre elemento p"; }

Elemento div

/*Evento: CLIC DERECHO DEL RATÓN*/
document.querySelector("div").oncontextmenu = function(e){
    e.preventDefault();
    document.querySelector("div").innerHTML = "Evento 'oncontextmenu' sobre elemento div";
}

Elemento h2

/*Evento: RATÓN DENTRO DEL ELEMENTO*/
document.querySelector("h2").onmouseenter = function(){
    document.querySelector("h2").innerHTML = "Evento 'onmouseenter' encima del elemento h2";
}

Elemento h3

/*Evento: RATÓN SOBRE EL ELEMENTO*/
document.querySelector("h3").onmouseover = function(){
    document.querySelector("h3").innerHTML = "Evento 'onmouseover' sobre elemento h3";
}
