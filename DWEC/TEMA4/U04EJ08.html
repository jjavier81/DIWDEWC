8. Realiza una web que muestre una imagen con un ancho de 150px y que permita hacer zoom hasta 1024px. El zoom se hará en incrementos de 25px, cuando el usuario pulse las teclas + y se deshará cuando pulse –.

onload = function() {
   onkeyup = zoom;
}

function zoom(e) {
   if (e.key == '+') {
     zoomin()
   }else if (e.key == '-') {
     zoomout()
   }
}

function zoomout(){
   var imgArale = document.getElementById("arale");
   var currWidth = imgArale.clientWidth;
   if(currWidth == 150){
     alert("Mínimo tamaño alcanzado.");
   } else {
     imgArale.style.width = (currWidth - 25) + "px";
   }
}

function zoomin(){
   var imgArale = document.getElementById("arale");
   var currWidth = imgArale.clientWidth;
   if(currWidth >= 1024){
     alert("Mínimo tamaño alcanzado.");
   } else{
     imgArale.style.width = (currWidth + 25) + "px";
   }
}
