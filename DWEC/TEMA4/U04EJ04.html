4. Utilizando el manejo de eventos del DOM 2, realiza una página que muestre una imagen de manera que esa imagen cambie cuando el ratón pase por encima de la imagen y que vuelva a cambiar cuando salga de ella.

addEventListener("DOMContentLoaded",function(){
  var imagen = document.getElementById('cambiaimagen');

    /*CAMBIAR LA IMAGEN CON EL RATÓN ENCIMA*/
  imagen.addEventListener("mouseover", change);
  function change(e){
    this.src = "https://pre00.deviantart.net/75c5/th/pre/f/2013/131/1/8/saint_seiya___seiya_hades_ending_render_by_thewolfmonster-d64w0nm.png";
  }

    /*VOLVER A LA IMAGEN PRINCIPAL CON EL RATÓN FUERA DE LA IMAGEN*/
  imagen.addEventListener("mouseout",function(e) {
    this.src = "https://vignette.wikia.nocookie.net/saintseiya/images/7/74/Seiya-Pegaso_161.JPG/revision/latest?cb=20141204214125&path-prefix=es";
  });

});
