onload = function() {
   onkeyup = cambiaMayuscula;
}

var mayuscula = false;

function cambiaMayuscula(e){
   var texto = document.getElementsByTagName('p')[1];
   if (e.key == 'Shift') {
     if (mayuscula) {
       texto.style.textTransform = 'lowercase';
       mayuscula = false;
     } else {
       texto.style.textTransform = 'uppercase';
       mayuscula = true;
     }
   }
}
