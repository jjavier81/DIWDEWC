    4. function Estudiante(nombre, apellidos, estudios, notamedia) {
      Persona.call(this,nombre, apellidos);
      this.estudios = estudios;
      this.notamedia = notamedia.toFixed(2);
      this.recuperar = function() {
        if (this.notamedia < 5) {
          this.notamedia = 5;
        }
      }
    }
